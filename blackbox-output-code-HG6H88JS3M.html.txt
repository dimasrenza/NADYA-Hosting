            const button = document.getElementById('noButton');

            const x = Math.random() * (rect.width - button.offsetWidth) + rect.left;
            const y = Math.random() * (rect.height - button.offsetHeight) + rect.top;
            button.style.left = x + 'px';
            button.style.top = y + 'px';
        });
